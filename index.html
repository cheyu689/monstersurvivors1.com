import pygame import random import sys # 初始化 Pygame pygame.init() # 定义颜色 WHITE = (255, 255, 255) BLACK = (0, 0, 0) GREEN = (0, 255, 0) RED = (255, 0, 0) BLUE = (0, 0, 255) GRAY = (200, 200, 200) # 设置窗口 WIDTH, HEIGHT = 600, 400 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Number Guessing Game") # 设置字体 title_font = pygame.font.SysFont("Arial", 32, bold=True) text_font = pygame.font.SysFont("Arial", 24) button_font = pygame.font.SysFont("Arial", 20) class Button: def __init__(self, x, y, width, height, text, color, hover_color): self.rect = pygame.Rect(x, y, width, height) self.text = text self.color = color self.hover_color = hover_color self.is_hovered = False def draw(self, surface): color = self.hover_color if self.is_hovered else self.color pygame.draw.rect(surface, color, self.rect, 0, 10) pygame.draw.rect(surface, BLACK, self.rect, 2, 10) text_surface = button_font.render(self.text, True, BLACK) text_rect = text_surface.get_rect(center=self.rect.center) surface.blit(text_surface, text_rect) def check_hover(self, pos): self.is_hovered = self.rect.collidepoint(pos) def is_clicked(self, pos, click): return self.rect.collidepoint(pos) and click class InputBox: def __init__(self, x, y, width, height): self.rect = pygame.Rect(x, y, width, height) self.text = "" self.active = False def handle_event(self, event): if event.type == pygame.MOUSEBUTTONDOWN: self.active = self.rect.collidepoint(event.pos) if event.type == pygame.KEYDOWN and self.active: if event.key == pygame.K_RETURN: return self.text elif event.key == pygame.K_BACKSPACE: self.text = self.text[:-1] else: # 只允许输入数字 if event.unicode.isdigit() and len(self.text) < 3: self.text += event.unicode return None def draw(self, surface): # 绘制输入框 color = BLUE if self.active else BLACK pygame.draw.rect(surface, WHITE, self.rect, 0, 5) pygame.draw.rect(surface, color, self.rect, 2, 5) # 渲染文本 text_surface = text_font.render(self.text, True, BLACK) surface.blit(text_surface, (self.rect.x + 10, self.rect.y + 5)) def guessing_game(): # 游戏变量 secret_number = random.randint(1, 100) max_attempts = 10 attempts = 0 message = "请猜一个1到100之间的数字" message_color = BLACK game_over = False restart_button = Button(200, 320, 200, 40, "重新开始", GREEN, (100, 255, 100)) guess_button = Button(350, 230, 150, 40, "提交猜测", BLUE, (100, 100, 255)) input_box = InputBox(150, 230, 150, 40) def restart_game(): nonlocal secret_number, attempts, message, message_color, game_over secret_number = random.randint(1, 100) attempts = 0 message = "请猜一个1到100之间的数字" message_color = BLACK game_over = False input_box.text = "" # 游戏主循环 running = True while running: mouse_pos = pygame.mouse.get_pos() mouse_click = False # 处理事件 for event in pygame.event.get(): if event.type == pygame.QUIT: running = False # 获取鼠标点击 if event.type == pygame.MOUSEBUTTONDOWN: if event.button == 1: mouse_click = True # 处理输入框 guess_text = input_box.handle_event(event) if guess_text and not game_over: try: guess = int(guess_text) if guess < 1 or guess > 100: message = "请输入1到100之间的数字" message_color = RED else: attempts += 1 if guess < secret_number: message = f"太小了！还剩 {max_attempts - attempts} 次机会" message_color = BLUE elif guess > secret_number: message = f"太大了！还剩 {max_attempts - attempts} 次机会" message_color = BLUE else: message = f"恭喜你！{attempts} 次猜中了！" message_color = GREEN game_over = True if attempts >= max_attempts and not game_over: message = f"游戏结束！正确答案是 {secret_number}" message_color = RED game_over = True input_box.text = "" except ValueError: message = "请输入有效的数字" message_color = RED # 检查按钮 restart_button.check_hover(mouse_pos) if restart_button.is_clicked(mouse_pos, mouse_click): restart_game() guess_button.check_hover(mouse_pos) if guess_button.is_clicked(mouse_pos, mouse_click) and not game_over: if input_box.text: try: guess = int(input_box.text) if guess < 1 or guess > 100: message = "请输入1到100之间的数字" message_color = RED else: attempts += 1 if guess < secret_number: message = f"太小了！还剩 {max_attempts - attempts} 次机会" message_color = BLUE elif guess > secret_number: message = f"太大了！还剩 {max_attempts - attempts} 次机会" message_color = BLUE else: message = f"恭喜你！{attempts} 次猜中了！" message_color = GREEN game_over = True if attempts >= max_attempts and not game_over: message = f"游戏结束！正确答案是 {secret_number}" message_color = RED game_over = True input_box.text = "" except ValueError: message = "请输入有效的数字" message_color = RED else: message = "请输入数字后再猜测" message_color = RED # 绘制界面 screen.fill(GRAY) # 绘制标题 title_text = title_font.render("猜数字游戏", True, BLACK) screen.blit(title_text, (WIDTH // 2 - title_text.get_width() // 2, 50)) # 绘制说明 info_text = text_font.render(f"猜一个1到100之间的数字，你有{max_attempts}次机会", True, BLACK) screen.blit(info_text, (WIDTH // 2 - info_text.get_width() // 2, 100)) # 绘制尝试次数 attempts_text = text_font.render(f"已尝试: {attempts}/{max_attempts}", True, BLACK) screen.blit(attempts_text, (WIDTH // 2 - attempts_text.get_width() // 2, 140)) # 绘制消息 message_text = text_font.render(message, True, message_color) screen.blit(message_text, (WIDTH // 2 - message_text.get_width() // 2, 180)) # 绘制输入提示 input_label = text_font.render("你的猜测:", True, BLACK) screen.blit(input_label, (50, 240)) # 绘制控件 input_box.draw(screen) guess_button.draw(screen) restart_button.draw(screen) # 更新显示 pygame.display.flip() pygame.quit() if __name__ == "__main__": guessing_game()
